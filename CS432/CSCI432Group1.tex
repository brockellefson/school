\section{Pseudo-Code(With Seth and Jake)}

\begin{algorithmic}
\STATE{Algorithm: Binary Search}
\STATE{Input: Desired Value to search for}
\STATE{Output: True or False}
\STATE{max = last element in array}
\STATE{min = 0}

\WHILE{min $<$= max}
\STATE{mid = max + min / 2}
    \IF{array[mid].value == desired value}
        \STATE return true
    \ELSIF{array[mid].value $<$ desired value}
        \STATE min = mid + 1
    \ELSE
        \STATE max = mid - 1
    \ENDIF
\ENDWHILE
\STATE{return false}
\end{algorithmic}
